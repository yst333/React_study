※ Fragments 이해
https://ko.reactjs.org/docs/fragments.html

1. Fragments 란?
React에서 컴포넌트가 여러 엘리먼트를 반환하는 것은 흔한 패턴입니다.
Fragments는 DOM에 별도의 노드를 추가하지 않고 여러 자식을 그룹화할 수 있습니다.

2. App_Fragments 소스 코딩

import React from 'react';
import './App.css';
import Fragments from './Fragments'

// 컴포넌트 단위로 element를 return할 때, 하나의 <html> 태그로 전체를 감싸지
// 않으면 에러가 발생합니다. 이때 <Fragments> 태그로 감싸면 불필요한
// <html> 태그를 추가하지 않고 사용할 수 있습니다.

function App() {
  return (
    <div>
      <h1>Start React 200!</h1>
      <p>Fragments 적용하기</p>
      <Fragments/>
    </div>
  );
}

export default App;

3. Fragments.js 소스 코딩

import React, {Component} from "react";

class Fragments extends Component {
    render(){
        return(            
            <React.Fragment>
                <p>P Tag 입니다!</p>
                <span>SPAN Tag 입니다!</span>
            </React.Fragment>  
        )
    }
}

export default Fragments;


4. App_Fragments_abb 소스 코딩

import React from 'react';
import './App.css';
import Fragments_abb from './Fragments_abb'

// 컴포넌트 단위로 element를 return할 때, 하나의 <html> 태그로 전체를 감싸지
// 않으면 에러가 발생합니다. 이때 <Fragments> 태그로 감싸면 불필요한
// <html> 태그를 추가하지 않고 사용할 수 있습니다.
// 이때, Fragments는 약식으로 표현도 가능하며, 약식 표현은
// <>로 시작해서 </>로 종결처리를 하면 됩니다.

function App() {
  return (
    <div>
      <h1>Start React 200!</h1>
      <p>Fragments 적용하기</p>
      <Fragments_abb/>
    </div>
  );
}

export default App;


5. Fragments_abb.js 소스 코딩

import React, {Component} from "react";

// Fragments는 약식으로 표현도 가능하며, 약식 표현은
// <>로 시작해서 </>로 종결처리를 하면 됩니다.
class Fragments_abb extends Component {
    render(){
        return(            
            <>
                <p>P Tag 입니다!</p>
                <span>SPAN Tag 입니다!</span>
            </>  
        )
    }
}

export default Fragments_abb;
